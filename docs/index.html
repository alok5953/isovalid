Quick Example

import { v } from '@alok5953/isovalid';

const userSchema = {
  username: v.string()
    .min(3)
    .max(20)
    .matches(/^[a-zA-Z0-9_]+$/),
  email: v.string()
    .email()
    .trimmed(),
  age: v.number()
    .min(18)
    .integer()
};

const result = userSchema.username.validate('john_doe');
console.log(result.valid); // true
